document.addEventListener("click", (event) => { const closeCircle = event.target.closest("#modal-close-circle"); if (!closeCircle) return; event.preventDefault(); event.stopPropagation(); closeModalPanel(); }); document.addEventListener("keydown", (event) => { if (event.key !== "Enter" && event.key !== " ") return; const closeCircle = event.target.closest("#modal-close-circle"); if (!closeCircle) return; event.preventDefault(); closeModalPanel(); });
